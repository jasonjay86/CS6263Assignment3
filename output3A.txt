    closest_match = 0

    return closest_match

def find_closest_binary_search(list, search_term):
    low = 0
    high = len(list) - 1
    while low <= high:
        mid = (low + high) // 2
        if list[mid] == search_term:
            return mid
        elif list[mid] < search_term:
            low = mid + 1
        else:

def insertion_sort(arr): 
    for i in range(1, len(arr)): 
        key = arr[i] 
        j = i-1
        while j >= 0 and key < arr[j] :
